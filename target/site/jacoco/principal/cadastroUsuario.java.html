 private usuarioDAO dao;    

    public void setUsuarioDAO(usuarioDAO dao) {
        this.dao = dao;
    }

    public void cadastrarUsuario() {
        String nome = txtNome.getText();
        String email = txtEmail.getText();
        String senha = txtSenha.getText();

        String tipo = "";
        if (rdbAdministrador.isSelected()) {
            tipo = "ADMINISTRADOR";
        } else if (rdbAtendente.isSelected()) {
            tipo = "ATENDENTE";
        }

        if (nome.isEmpty() || senha.isEmpty() || email.isEmpty() || tipo.isEmpty()) {
            JOptionPane.showMessageDialog(null, "Por favor, preencha todos os campos e selecione um cargo!");
            return;
        }

        if (!validarEmail(email)) {
            JOptionPane.showMessageDialog(null, "Por favor, insira um e-mail válido!");
            return;
        }

        UsuarioDados usrDados = new UsuarioDados();
        usrDados.setNome_usuario(nome);
        usrDados.setEmail_usuario(email);
        usrDados.setSenha_usuario(senha);
        usrDados.setTipo_usuario(tipo);

        // Usa o DAO injetado ou cria um novo (caso não esteja testando)
        if (dao == null) {
            dao = new usuarioDAO();
        }

        boolean cadastrou = dao.cadastrarUsuario(usrDados);

        if (cadastrou) {
            JOptionPane.showMessageDialog(null, "Usuário cadastrado com sucesso!");
            dispose(); // ou qualquer ação pós-cadastro
        }
    }

    private boolean validarEmail(String email) {
        String regex = "^[A-Za-z0-9+_.-]+@(gmail\\.com|hotmail\\.com|outlook\\.com|yahoo\\.com|icloud\\.com|protonmail\\.com|live\\.com|bol\\.com|uol\\.com\\.br|terra\\.com\\.br|yandex\\.com|mail\\.ru)$";
        return email.matches(regex);
    }
