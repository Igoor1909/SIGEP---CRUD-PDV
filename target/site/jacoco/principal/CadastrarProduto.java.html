    public void cadastrarProduto() {
        String nome = txtNome.getText();
        String valorStr = txtValor.getText();
        String ean = txtEAN.getText();

        if (nome.isEmpty() || valorStr.isEmpty() || ean.isEmpty()) {
            JOptionPane.showMessageDialog(null, "Por favor, preencha todos os campos!");
            return;
        }

        if (valorStr.contains(".")) {
            JOptionPane.showMessageDialog(null, "Utilize vírgula (,) como separador decimal, e não ponto (.)!");
            return;
        }

        try {
            double valor = Double.parseDouble(valorStr.replace(",", "."));

            ProdutoDados objProduto = new ProdutoDados();
            objProduto.setNome_produto(nome);
            objProduto.setValor(valor);
            objProduto.setEan_produto(ean);

            produtoDAO.cadastrarProduto(objProduto);  // usa atributo, não cria novo
            limparTela();
        } catch (NumberFormatException e) {
            JOptionPane.showMessageDialog(null, "Valor do produto inválido. Certifique-se de que o campo contém apenas números e vírgula.");
        }
    }

    public void limparTela() {
        txtNome.setText("");
        txtValor.setText("");
        txtEAN.setText("");
        txtNome.requestFocus();
    }
