    private void executar() {
        if (txtID.getText().isEmpty()) {
            JOptionPane.showMessageDialog(null, "Por favor, preencha o campo ID");
        } else {
            int id = Integer.parseInt(txtID.getText());
            ProdutoDados a = new produtoDAO().getProdutoDados(id);

            if (a != null) {
                txtNomeProduto.setText(a.getNome_produto());
                txtValorProduto.setText(String.valueOf(a.getValor()));
            } else {
                JOptionPane.showMessageDialog(null, "Registro não encontrado.");
            }
        }
    }

    private void alterarCliente() {

        
        String nome = txtNomeProduto.getText();
        String valor = txtValorProduto.getText();

        if (txtID.getText().isEmpty()) {
            JOptionPane.showMessageDialog(null, "Por favor, preencha o campo ID");
        } else if (nome.isEmpty() || valor.isEmpty()) {
            JOptionPane.showMessageDialog(null, "Por favor, preencha todos os campos!");
        } else {
            int id_Cliente = Integer.parseInt(txtID.getText());
            ProdutoDados objProduto = new ProdutoDados();
            objProduto.setId_produto(id_Cliente);
            objProduto.setNome_produto(nome);
            objProduto.setValor(Double.parseDouble(valor));

            new produtoDAO().alterarProduto(objProduto);
            JOptionPane.showMessageDialog(null, "Produto alterado!");
            limpar();
        }

    }

    private void excluirProduto() {

        if (txtID.getText().isEmpty()) {
            JOptionPane.showMessageDialog(null, "Por favor, preencha o campo ID");
        } else {
            int id_Produto = Integer.parseInt(txtID.getText());
            ProdutoDados objProduto = new ProdutoDados();
            objProduto.setId_produto(id_Produto);

            new produtoDAO().excluirProduto(objProduto);

            limpar();

        }
    }

    private void limpar() {

        txtID.setText("");
        txtNomeProduto.setText("");
        txtValorProduto.setText("");

        txtID.requestFocus();
    }
