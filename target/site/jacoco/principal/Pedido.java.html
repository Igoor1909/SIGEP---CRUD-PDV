    public void ExecutarProduto(JTextField txtIdProduto, JTextField txtNomeProduto, JTextField txtValorProduto, JTextField txtQuantidadeProduto) {
        String idText = txtIdProduto.getText();

        if (idText.isEmpty()) {
            JOptionPane.showMessageDialog(null, "Por favor, preencha o campo ID");
            return;
        }

        int id;
        try {
            id = Integer.parseInt(idText);
        } catch (NumberFormatException e) {
            JOptionPane.showMessageDialog(null, "Por favor, insira um valor numérico para o ID");
            return;
        }

        ProdutoDados a = new produtoDAO().getProdutoDados(id);

        if (a != null) {
            txtNomeProduto.setText(a.getNome_produto());
            txtValorProduto.setText(String.valueOf(a.getValor()));
            txtQuantidadeProduto.setText("0");
        } else {
            JOptionPane.showMessageDialog(null, "Registro não encontrado.");
        }
    }

    public void txtIdProdutoActionPerformed(JTextField txtIdProduto, JTextField txtNomeProduto, JTextField txtValorProduto, JTextField txtQuantidadeProduto) {
        txtIdProduto.addActionListener(e -> ExecutarProduto(txtIdProduto, txtNomeProduto, txtValorProduto, txtQuantidadeProduto));

        txtIdProduto.addKeyListener(new KeyAdapter() {
            @Override
            public void keyPressed(KeyEvent e) {
                if (e.getKeyCode() == KeyEvent.VK_ENTER) {
                    ExecutarProduto(txtIdProduto, txtNomeProduto, txtValorProduto, txtQuantidadeProduto);
                }
            }
        });
    }




    public void calcularTotal() {

        try {
            String valorProdutoStr = txtValorProduto.getText();
            String quantidadeProdutoStr = txtQuantidadeProduto.getText();
            String nomeClienteStr = txtNomeCliente.getText();
            String nomeProdutoStr = txtNomeProduto.getText();
            String formaPagamentoStr = (String) cbxFormasPagamento.getSelectedItem();
            String dataRetiradaStr = txtRetirada.getText();

            if (valorProdutoStr.isEmpty() || quantidadeProdutoStr.isEmpty() || nomeClienteStr.isEmpty() || nomeProdutoStr.isEmpty() || formaPagamentoStr.isEmpty() || dataRetiradaStr.isEmpty()) {
                JOptionPane.showMessageDialog(null, "Por favor! Preencha todos os componentes do pedido");
            } else {
                double valorProduto = 0.0;
                int quantidadeProduto = 0;
                valorProduto = Double.parseDouble(valorProdutoStr);
                quantidadeProduto = Integer.parseInt(quantidadeProdutoStr);

                if (valorProduto == 0.0 || quantidadeProduto == 0) {
                    JOptionPane.showMessageDialog(null, "Valor e quantidade do produto não podem ser 0!");
                } else {
                    double valorTotal = valorProduto * quantidadeProduto;

                    lblValorTotal.setText(String.valueOf(valorTotal));
                    lblCliente.setText(nomeClienteStr);
                    lblProduto.setText(nomeProdutoStr);
                    lblQuantidade.setText(quantidadeProdutoStr);
                    lblFormaPagamento.setText(formaPagamentoStr);
                    lblRetirada.setText(dataRetiradaStr);
                }

            }

        } catch (Exception e) {
            JOptionPane.showMessageDialog(null, "Por favor, insira valores numéricos válidos.");

        }
    }

    public void RealizarPedido(){
        if(lblCliente.getText().isEmpty() || lblProduto.getText().isEmpty() || lblQuantidade.getText().isEmpty() || lblValorTotal.getText().isEmpty() || lblFormaPagamento.getText().isEmpty() || lblRetirada.getText().isEmpty()){
            JOptionPane.showMessageDialog(null, "Por favor! Calcule os valores antes de confirmar o pedido");
        }else{
            PedidoDados objPedido = new PedidoDados();
            objPedido.setId_cliente(Integer.parseInt(txtIdCliente.getText()));
            objPedido.setId_produto(Integer.parseInt(txtIdProduto.getText()));
            objPedido.setQuantidade(Integer.parseInt(lblQuantidade.getText()));
            objPedido.setValor_unitario(Double.parseDouble(txtValorProduto.getText()));
            objPedido.setValor_total(Double.parseDouble(lblValorTotal.getText()));
            objPedido.setForma_pagamento(lblFormaPagamento.getText());
            objPedido.setData_retirada(txtRetirada.getText());
            objPedido.setStatus("Em aberto");
            
            new PedidoDAO().RealizarPedido(objPedido);
            JOptionPane.showMessageDialog(null, "Pedido Realizado com sucesso!");
            LimparDados();
             
        }
    }
