    private FornecedorDAO fornecedorDAO;

// Setter para injetar o DAO (usado nos testes para injetar mock)
    public void setFornecedorDAO(FornecedorDAO fornecedorDAO) {
        this.fornecedorDAO = fornecedorDAO;
    }

    public void cadastrarFornecedor() {
        // Validação dos campos obrigatórios
        if (txtNomeFornecedor.getText().isEmpty() || txtCNPJFornecedor.getText().isEmpty()
                || txtRuaFornecedor.getText().isEmpty() || txtNumeroFornecedor.getText().isEmpty()
                || txtCidadeFornecedor.getText().isEmpty() || txtEstadoFornecedor.getText().isEmpty()
                || txtBairroFornecedor.getText().isEmpty()) {
            JOptionPane.showMessageDialog(this, "Por favor, preencha todos os campos obrigatórios!");
            return;
        }

        // Se o DAO não foi injetado, cria uma instância real
        if (fornecedorDAO == null) {
            fornecedorDAO = new FornecedorDAO();
        }

        try {
            String cnpj = txtCNPJFornecedor.getText().trim();

            // Valida se o CNPJ contém apenas números
            if (!cnpj.matches("\\d+")) {
                JOptionPane.showMessageDialog(this, "CNPJ deve conter apenas números!", "Erro de Formato", JOptionPane.ERROR_MESSAGE);
                txtCNPJFornecedor.requestFocus();
                return;
            }

            String estado = txtEstadoFornecedor.getText().trim();

            // Valida estado completo
            if (!validarEstadoCompleto(estado)) {
                JOptionPane.showMessageDialog(this,
                        "Estado inválido! Digite o nome completo correto (ex: São Paulo, Minas Gerais).\n"
                        + "Obs: Use acentos quando necessário (ex: Pará, Amapá).",
                        "Erro de Formato", JOptionPane.ERROR_MESSAGE);
                txtEstadoFornecedor.requestFocus();
                return;
            }

            // Preenche objeto Fornecedor
            FornecedorDados objFornecedor = new FornecedorDados();
            objFornecedor.setNome_fornecedor(txtNomeFornecedor.getText().trim());
            objFornecedor.setCnpj_fornecedor(cnpj);
            objFornecedor.setRua_fornecedor(txtRuaFornecedor.getText().trim());
            objFornecedor.setNumero_fornecedor(Integer.parseInt(txtNumeroFornecedor.getText().trim()));
            objFornecedor.setComplemento_fornecedor(txtComplementoFornecedor.getText().trim());
            objFornecedor.setCidade_fornecedor(txtCidadeFornecedor.getText().trim());
            objFornecedor.setEstado_fornecedor(estado);
            objFornecedor.setBairro_fornecedor(txtBairroFornecedor.getText().trim());

            // Cadastra e limpa tela
            fornecedorDAO.cadastrarFornecedor(objFornecedor);
            limparTela();

            // Mensagem final de sucesso
            JOptionPane.showMessageDialog(this, "Fornecedor cadastrado com sucesso!");

        } catch (NumberFormatException e) {
            JOptionPane.showMessageDialog(this, "Número do endereço deve conter apenas valores numéricos!",
                    "Erro de Formato", JOptionPane.ERROR_MESSAGE);
        } catch (Exception e) {
            JOptionPane.showMessageDialog(this, "Erro ao cadastrar fornecedor: " + e.getMessage(),
                    "Erro", JOptionPane.ERROR_MESSAGE);
        }
    }

    public boolean validarEstadoCompleto(String estado) {
        String[] estadosBrasileiros = {
            "Acre", "Alagoas", "Amapá", "Amazonas", "Bahia", "Ceará",
            "Distrito Federal", "Espírito Santo", "Goiás", "Maranhão",
            "Mato Grosso", "Mato Grosso do Sul", "Minas Gerais", "Pará",
            "Paraíba", "Paraná", "Pernambuco", "Piauí", "Rio de Janeiro",
            "Rio Grande do Norte", "Rio Grande do Sul", "Rondônia", "Roraima",
            "Santa Catarina", "São Paulo", "Sergipe", "Tocantins"
        };

        for (String nomeEstado : estadosBrasileiros) {
            if (nomeEstado.equalsIgnoreCase(estado.trim())) {
                return true;
            }
        }
        return false;
    }
